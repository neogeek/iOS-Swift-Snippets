# Logic

## If Statements

```swift
let scale:Int = 8

if scale < 10 {
    print("Less than 10")
}
```

```swift
let hat:String = "Fez"

if hat == "Fedora" {
    print("Hat is a Fedora.")
} else {
    print("Hat is not a Fedora.")
}
```

```swift
let hat:String = "Fez"

if !hat.isEmpty {
    print("You are wearing a hat")
}
```
